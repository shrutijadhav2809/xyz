#include <stdio.h>

int main() {
    int n, f;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    int pages[n];
    printf("Enter page reference string: ");
    for(int i = 0; i < n; i++) {
        scanf("%d", &pages[i]);
    }

    printf("Enter number of frames: ");
    scanf("%d", &f);

    int frame[f], time[f];
    int page_faults = 0;

    // Initialize frames
    for(int i = 0; i < f; i++) {
        frame[i] = -1;
        time[i] = -1;
    }

    for(int i = 0; i < n; i++) {
        int found = 0;

        // Check if page already in frame
        for(int j = 0; j < f; j++) {
            if(frame[j] == pages[i]) {
                found = 1;
                time[j] = i;   // update recent use
                break;
            }
        }

        // If page not found → page fault
        if(found == 0) {
            int pos = 0;

            // Find least recently used page
            for(int j = 1; j < f; j++) {
                if(time[j] < time[pos]) {
                    pos = j;
                }
            }

            frame[pos] = pages[i];
            time[pos] = i;
            page_faults++;
        }

        // Print current frames (optional but good for viva)
        printf("Frames: ");
        for(int j = 0; j < f; j++) {
            if(frame[j] != -1)
                printf("%d ", frame[j]);
            else
                printf("- ");
        }
        printf("\n");
    }

    printf("\nTotal Page Faults = %d\n", page_faults);

    return 0;
}
