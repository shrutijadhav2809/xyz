#include <stdio.h>

int main() {
    int n;

    printf("Enter number of disk requests: ");
    scanf("%d", &n);

    int req[n];

    printf("Enter request queue: ");
    for(int i = 0; i < n; i++) {
        scanf("%d", &req[i]);
    }

    int head;
    printf("Enter initial head position: ");
    scanf("%d", &head);

    int direction;
    printf("Enter direction (1 = right, 0 = left): ");
    scanf("%d", &direction);

    int total_seek = 0;

    // Sort the requests (bubble sort)
    for(int i = 0; i < n-1; i++) {
        for(int j = 0; j < n-i-1; j++) {
            if(req[j] > req[j+1]) {
                int temp = req[j];
                req[j] = req[j+1];
                req[j+1] = temp;
            }
        }
    }

    int index;

    // Find position of head
    for(int i = 0; i < n; i++) {
        if(head < req[i]) {
            index = i;
            break;
        }
    }

    printf("\nSeek Sequence: %d ", head);

    if(direction == 1) {
        // Move right first
        for(int i = index; i < n; i++) {
            int diff = (head > req[i]) ? (head - req[i]) : (req[i] - head);
            total_seek += diff;
            head = req[i];
            printf("-> %d ", head);
        }

        // Go to disk end (assume max = 199)
        total_seek += (199 - head);
        head = 199;
        printf("-> %d ", head);

        // Reverse (move left)
        for(int i = index-1; i >= 0; i--) {
            int diff = (head > req[i]) ? (head - req[i]) : (req[i] - head);
            total_seek += diff;
            head = req[i];
            printf("-> %d ", head);
        }
    } 
    else {
        // Move left first
        for(int i = index-1; i >= 0; i--) {
            int diff = (head > req[i]) ? (head - req[i]) : (req[i] - head);
            total_seek += diff;
            head = req[i];
            printf("-> %d ", head);
        }

        // Go to disk start (0)
        total_seek += head;
        head = 0;
        printf("-> %d ", head);

        // Reverse (move right)
        for(int i = index; i < n; i++) {
            int diff = (head > req[i]) ? (head - req[i]) : (req[i] - head);
            total_seek += diff;
            head = req[i];
            printf("-> %d ", head);
        }
    }

    printf("\n\nTotal Seek Time = %d\n", total_seek);

    return 0;
}
